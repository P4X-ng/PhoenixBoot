5/// Allows configuring the bootloader behavior.
6///
7/// TODO: describe use together with `entry_point` macro
8/// TODO: example
9#[derive(Debug, PartialEq, Eq, Clone, Copy)]
10#[non_exhaustive]
11pub struct BootloaderConfig {
12    /// The version of the bootloader API.
13    ///
14    /// Automatically generated from the crate version. Checked on deserialization to
15    /// ensure that the kernel and bootloader use the same API version, i.e. the same config
16    /// and boot info format.
17    pub(crate) version: ApiVersion,
18
19    /// Configuration for (optional) page table mappings created by the bootloader.
20    pub mappings: Mappings,
21
22    /// The size of the stack that the bootloader should allocate for the kernel (in bytes).
23    ///
24    /// The bootloader starts the kernel with a valid stack pointer. This setting defines
25    /// the stack size that the bootloader should allocate and map.
26    ///
27    /// The stack is created with a additional guard page, so a stack overflow will lead to
28    /// a page fault.
29    pub kernel_stack_size: u64,
30
31    /// Configuration for the frame buffer that can be used by the kernel to display pixels
32    /// on the screen.
33    #[deprecated(
34        since = "0.11.1",
35        note = "The frame buffer is now configured through the `BootConfig` struct when creating the bootable disk image"
36    )]
37    pub frame_buffer: FrameBuffer,
38}
39
40impl BootloaderConfig {
41    pub(crate) const UUID: [u8; 16] = [
42        0x74, 0x3C, 0xA9, 0x61, 0x09, 0x36, 0x46, 0xA0, 0xBB, 0x55, 0x5C, 0x15, 0x89, 0x15, 0x25,
43        0x3D,
44    ];
45    #[doc(hidden)]
46    pub const SERIALIZED_LEN: usize = 133;
47
48    /// Creates a new default configuration with the following values:
49    ///
50    /// - `kernel_stack_size`: 80kiB
51    /// - `mappings`: See [`Mappings::new_default()`]
52    pub const fn new_default() -> Self {
53        Self {
54            kernel_stack_size: 80 * 1024,
55            version: ApiVersion::new_default(),
56            mappings: Mappings::new_default(),
57            frame_buffer: FrameBuffer::new_default(),
58        }
59    }
60
61    /// Serializes the configuration to a byte array.
62    ///
63    /// This is used by the [`crate::entry_point`] macro to store the configuration in a
64    /// dedicated section in the resulting ELF file.
65    pub const fn serialize(&self) -> [u8; Self::SERIALIZED_LEN] {
66        let Self {
67            version,
68            mappings,
69            kernel_stack_size,
70            frame_buffer,
71        } = self;
72        let ApiVersion {
73            version_major,
74            version_minor,
75            version_patch,
76            pre_release,
77        } = version;
78        let Mappings {
79            kernel_stack,
80            kernel_base,
81            boot_info,
82            framebuffer,
83            physical_memory,
84            page_table_recursive,
85            aslr,
86            dynamic_range_start,
87            dynamic_range_end,
88            ramdisk_memory,
89        } = mappings;
90        let FrameBuffer {
91            minimum_framebuffer_height,
92            minimum_framebuffer_width,
93        } = frame_buffer;
94
95        let version = {
96            let one = concat_2_2(version_major.to_le_bytes(), version_minor.to_le_bytes());
97            let two = concat_2_1(version_patch.to_le_bytes(), [*pre_release as u8]);
98            concat_4_3(one, two)
99        };
100        let buf = concat_16_7(Self::UUID, version);
101
102        let buf = concat_23_8(buf, kernel_stack_size.to_le_bytes());
103
104        let buf = concat_31_9(buf, kernel_stack.serialize());
105        let buf = concat_40_9(buf, kernel_base.serialize());
106
107        let buf = concat_49_9(buf, boot_info.serialize());
108        let buf = concat_58_9(buf, framebuffer.serialize());
109
110        let buf = concat_67_10(
111            buf,
112            match physical_memory {
113                Option::None => [0; 10],
114                Option::Some(m) => concat_1_9([1], m.serialize()),
115            },
116        );
117        let buf = concat_77_10(
118            buf,
119            match page_table_recursive {
120                Option::None => [0; 10],
121                Option::Some(m) => concat_1_9([1], m.serialize()),
122            },
123        );
124        let buf = concat_87_1(buf, [(*aslr) as u8]);
125        let buf = concat_88_9(
126            buf,
127            match dynamic_range_start {
128                Option::None => [0; 9],
129                Option::Some(addr) => concat_1_8([1], addr.to_le_bytes()),
130            },
131        );
132        let buf = concat_97_9(
133            buf,
134            match dynamic_range_end {
135                Option::None => [0; 9],
136                Option::Some(addr) => concat_1_8([1], addr.to_le_bytes()),
137            },
138        );
139
140        let buf = concat_106_9(buf, ramdisk_memory.serialize());
141
142        let buf = concat_115_9(
143            buf,
144            match minimum_framebuffer_height {
145                Option::None => [0; 9],
146                Option::Some(addr) => concat_1_8([1], addr.to_le_bytes()),
147            },
148        );
149
150        concat_124_9(
151            buf,
152            match minimum_framebuffer_width {
153                Option::None => [0; 9],
154                Option::Some(addr) => concat_1_8([1], addr.to_le_bytes()),
155            },
156        )
157    }
158
159    /// Tries to deserialize a config byte array that was created using [`Self::serialize`].
160    ///
161    /// This is used by the bootloader to deserialize the configuration given in the kernel's
162    /// ELF file.
163    ///
164    /// TODO: return error enum
165    pub fn deserialize(serialized: &[u8]) -> Result<Self, &'static str> {
166        if serialized.len() != Self::SERIALIZED_LEN {
167            return Err("invalid len");
168        }
169
170        let s = serialized;
171
172        let (uuid, s) = split_array_ref(s);
173        if uuid != &Self::UUID {
174            return Err("invalid UUID");
175        }
176
177        let (version, s) = {
178            let (&major, s) = split_array_ref(s);
179            let (&minor, s) = split_array_ref(s);
180            let (&patch, s) = split_array_ref(s);
181            let (&pre, s) = split_array_ref(s);
182            let pre = match pre {
183                [0] => false,
184                [1] => true,
185                _ => return Err("invalid pre version"),
186            };
187
188            let version = ApiVersion {
189                version_major: u16::from_le_bytes(major),
190                version_minor: u16::from_le_bytes(minor),
191                version_patch: u16::from_le_bytes(patch),
192                pre_release: pre,
193            };
194            (version, s)
195        };
196
197        // TODO check version against this crate version -> error if they're different
198
199        let (&kernel_stack_size, s) = split_array_ref(s);
200
201        let (mappings, s) = {
202            let (&kernel_stack, s) = split_array_ref(s);
203            let (&kernel_base, s) = split_array_ref(s);
204            let (&boot_info, s) = split_array_ref(s);
205            let (&framebuffer, s) = split_array_ref(s);
206            let (&physical_memory_some, s) = split_array_ref(s);
207            let (&physical_memory, s) = split_array_ref(s);
208            let (&page_table_recursive_some, s) = split_array_ref(s);
209            let (&page_table_recursive, s) = split_array_ref(s);
210            let (&[alsr], s) = split_array_ref(s);
211            let (&dynamic_range_start_some, s) = split_array_ref(s);
212            let (&dynamic_range_start, s) = split_array_ref(s);
213            let (&dynamic_range_end_some, s) = split_array_ref(s);
214            let (&dynamic_range_end, s) = split_array_ref(s);
215            let (&ramdisk_memory, s) = split_array_ref(s);
216
217            let mappings = Mappings {
218                kernel_stack: Mapping::deserialize(&kernel_stack)?,
219                kernel_base: Mapping::deserialize(&kernel_base)?,
220                boot_info: Mapping::deserialize(&boot_info)?,
221                framebuffer: Mapping::deserialize(&framebuffer)?,
222                physical_memory: match physical_memory_some {
223                    [0] if physical_memory == [0; 9] => Option::None,
224                    [1] => Option::Some(Mapping::deserialize(&physical_memory)?),
225                    _ => return Err("invalid phys memory value"),
226                },
227                page_table_recursive: match page_table_recursive_some {
228                    [0] if page_table_recursive == [0; 9] => Option::None,
229                    [1] => Option::Some(Mapping::deserialize(&page_table_recursive)?),
230                    _ => return Err("invalid page table recursive value"),
231                },
232                aslr: match alsr {
233                    1 => true,
234                    0 => false,
235                    _ => return Err("invalid aslr value"),
236                },
237                dynamic_range_start: match dynamic_range_start_some {
238                    [0] if dynamic_range_start == [0; 8] => Option::None,
239                    [1] => Option::Some(u64::from_le_bytes(dynamic_range_start)),
240                    _ => return Err("invalid dynamic range start value"),
241                },
242                dynamic_range_end: match dynamic_range_end_some {
243                    [0] if dynamic_range_end == [0; 8] => Option::None,
244                    [1] => Option::Some(u64::from_le_bytes(dynamic_range_end)),
245                    _ => return Err("invalid dynamic range end value"),
246                },
247                ramdisk_memory: Mapping::deserialize(&ramdisk_memory)?,
248            };
249            (mappings, s)
250        };
251
252        let (frame_buffer, s) = {
253            let (&min_framebuffer_height_some, s) = split_array_ref(s);
254            let (&min_framebuffer_height, s) = split_array_ref(s);
255            let (&min_framebuffer_width_some, s) = split_array_ref(s);
256            let (&min_framebuffer_width, s) = split_array_ref(s);
257
258            let frame_buffer = FrameBuffer {
259                minimum_framebuffer_height: match min_framebuffer_height_some {
260                    [0] if min_framebuffer_height == [0; 8] => Option::None,
261                    [1] => Option::Some(u64::from_le_bytes(min_framebuffer_height)),
262                    _ => return Err("minimum_framebuffer_height invalid"),
263                },
264                minimum_framebuffer_width: match min_framebuffer_width_some {
265                    [0] if min_framebuffer_width == [0; 8] => Option::None,
266                    [1] => Option::Some(u64::from_le_bytes(min_framebuffer_width)),
267                    _ => return Err("minimum_framebuffer_width invalid"),
268                },
269            };
270            (frame_buffer, s)
271        };
272
273        if !s.is_empty() {
274            return Err("unexpected rest");
275        }
276
277        Ok(Self {
278            version,
279            kernel_stack_size: u64::from_le_bytes(kernel_stack_size),
280            mappings,
281            frame_buffer,
282        })
283    }
284
285    #[cfg(test)]
286    fn random() -> Self {
287        Self {
288            version: ApiVersion::random(),
289            mappings: Mappings::random(),
290            kernel_stack_size: rand::random(),
291            frame_buffer: FrameBuffer::random(),
292        }
293    }
294}

366/// Allows to configure the virtual memory mappings created by the bootloader.
367#[derive(Debug, Default, PartialEq, Eq, Clone, Copy)]
368#[non_exhaustive]
369pub struct Mappings {
370    /// Configures how the kernel stack should be mapped.
371    ///
372    /// If a fixed address is set, it must be page aligned.
373    ///
374    /// Note that the first page of the kernel stack is intentionally left unmapped
375    /// to act as a guard page. This ensures that a page fault occurs on a stack
376    /// overflow. For example, setting the kernel stack address to
377    /// `FixedAddress(0xf_0000_0000)` will result in a guard page at address
378    /// `0xf_0000_0000` and the kernel stack starting at address `0xf_0000_1000`.
379    pub kernel_stack: Mapping,
380    /// Configures the base address of the kernel.
381    ///
382    /// If a fixed address is set, it must be paged aligned and the kernel must be
383    /// a position-independent exectuable.
384    pub kernel_base: Mapping,
385    /// Specifies where the [`crate::BootInfo`] struct should be placed in virtual memory.
386    pub boot_info: Mapping,
387    /// Specifies the mapping of the frame buffer memory region.
388    pub framebuffer: Mapping,
389    /// The bootloader supports mapping the whole physical memory into the virtual address
390    /// space at some offset. This is useful for accessing and modifying the page tables set
391    /// up by the bootloader.
392    ///
393    /// This mapping will go from physical address `0x0` to whichever is larger:
394    /// - The end of the last region in the BIOS/UEFI memory map
395    /// - The address `0x1_0000_0000` (such that at least 4 GiB of physical memory are always mapped).
396    ///   This is to ensure that useful MMIO regions (local APIC, I/O APIC, PCI bars) are
397    ///   accessible to the kernel even if less physical memory than that is on the system.
398    ///
399    /// Defaults to `None`, i.e. no mapping of the physical memory.
400    pub physical_memory: Option<Mapping>,
401    /// As an alternative to mapping the whole physical memory (see [`Self::physical_memory`]),
402    /// the bootloader also has support for setting up a
403    /// [recursive level 4 page table](https://os.phil-opp.com/paging-implementation/#recursive-page-tables).
404    ///
405    /// Defaults to `None`, i.e. no recursive mapping.
406    pub page_table_recursive: Option<Mapping>,
407    /// Whether to randomize non-statically configured addresses.
408    /// The kernel base address will be randomized when it's compiled as
409    /// a position independent executable.
410    ///
411    /// Defaults to `false`.
412    pub aslr: bool,
413    /// The lowest virtual address for dynamic addresses.
414    ///
415    /// Defaults to `0`.
416    pub dynamic_range_start: Option<u64>,
417    /// The highest virtual address for dynamic addresses.
418    ///
419    /// Defaults to `0xffff_ffff_ffff_f000`.
420    pub dynamic_range_end: Option<u64>,
421    /// Virtual address to map ramdisk image, if present on disk
422    /// Defaults to dynamic
423    pub ramdisk_memory: Mapping,
424}
425
426impl Mappings {
427    /// Creates a new mapping configuration with dynamic mapping for kernel, boot info and
428    /// frame buffer. Neither physical memory mapping nor recursive page table creation are
429    /// enabled.
430    pub const fn new_default() -> Self {
431        Self {
432            kernel_stack: Mapping::new_default(),
433            kernel_base: Mapping::new_default(),
434            boot_info: Mapping::new_default(),
435            framebuffer: Mapping::new_default(),
436            physical_memory: Option::None,
437            page_table_recursive: Option::None,
438            aslr: false,
439            dynamic_range_start: None,
440            dynamic_range_end: None,
441            ramdisk_memory: Mapping::new_default(),
442        }
443    }
444
445    #[cfg(test)]
446    fn random() -> Mappings {
447        let phys = rand::random();
448        let recursive = rand::random();
449        Self {
450            kernel_stack: Mapping::random(),
451            kernel_base: Mapping::random(),
452            boot_info: Mapping::random(),
453            framebuffer: Mapping::random(),
454            physical_memory: if phys {
455                Option::Some(Mapping::random())
456            } else {
457                Option::None
458            },
459            page_table_recursive: if recursive {
460                Option::Some(Mapping::random())
461            } else {
462                Option::None
463            },
464            aslr: rand::random(),
465            dynamic_range_start: if rand::random() {
466                Option::Some(rand::random())
467            } else {
468                Option::None
469            },
470            dynamic_range_end: if rand::random() {
471                Option::Some(rand::random())
472            } else {
473                Option::None
474            },
475            ramdisk_memory: Mapping::random(),
476        }
477    }
478}

480/// Specifies how the bootloader should map a memory region into the virtual address space.
481#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
482pub enum Mapping {
483    /// Look for an unused virtual memory region at runtime.
484    Dynamic,
485    /// Try to map the region at the given virtual address.
486    ///
487    /// The given virtual address must be page-aligned.
488    ///
489    /// This setting can lead to runtime boot errors if the given address is not aligned,
490    /// already in use, or invalid for other reasons.
491    FixedAddress(u64),
492}
493
494impl Mapping {
495    /// Creates a new [`Mapping::Dynamic`].
496    ///
497    /// This function has identical results as [`Default::default`], the only difference is
498    /// that this is a `const` function.
499    pub const fn new_default() -> Self {
500        Self::Dynamic
501    }
502
503    #[cfg(test)]
504    fn random() -> Mapping {
505        let fixed = rand::random();
506        if fixed {
507            Self::Dynamic
508        } else {
509            Self::FixedAddress(rand::random())
510        }
511    }
512
513    const fn serialize(&self) -> [u8; 9] {
514        match self {
515            Mapping::Dynamic => [0; 9],
516            Mapping::FixedAddress(addr) => concat_1_8([1], addr.to_le_bytes()),
517        }
518    }
519
520    fn deserialize(serialized: &[u8; 9]) -> Result<Self, &'static str> {
521        let (&variant, s) = split_array_ref(serialized);
522        let (&addr, s) = split_array_ref(s);
523        if !s.is_empty() {
524            return Err("invalid mapping format");
525        }
526
527        match variant {
528            [0] if addr == [0; 8] => Ok(Mapping::Dynamic),
529            [1] => Ok(Mapping::FixedAddress(u64::from_le_bytes(addr))),
530            _ => Err("invalid mapping value"),
531        }
532    }
533}
534
535impl Default for Mapping {
536    fn default() -> Self {
537        Self::new_default()
538    }
539}

541/// Configuration for the frame buffer used for graphical output.
542#[derive(Debug, Default, PartialEq, Eq, Clone, Copy)]
543#[non_exhaustive]
544pub struct FrameBuffer {
545    /// Instructs the bootloader to set up a framebuffer format that has at least the given height.
546    ///
547    /// If this is not possible, the bootloader will fall back to a smaller format.
548    pub minimum_framebuffer_height: Option<u64>,
549    /// Instructs the bootloader to set up a framebuffer format that has at least the given width.
550    ///
551    /// If this is not possible, the bootloader will fall back to a smaller format.
552    pub minimum_framebuffer_width: Option<u64>,
553}
554

557    pub const fn new_default() -> Self {
558        Self {
559            minimum_framebuffer_height: Option::None,
560            minimum_framebuffer_width: Option::None,
561        }
562    }
563
564    #[cfg(test)]
565    fn random() -> FrameBuffer {
566        Self {
567            minimum_framebuffer_height: if rand::random() {
568                Option::Some(rand::random())
569            } else {
570                Option::None
571            },
572            minimum_framebuffer_width: if rand::random() {
573                Option::Some(rand::random())
574            } else {
575                Option::None
576            },
577        }
578    }

581/// Taken from https://github.com/rust-lang/rust/blob/e100ec5bc7cd768ec17d75448b29c9ab4a39272b/library/core/src/slice/mod.rs#L1673-L1677
582///
583/// TODO replace with `split_array` feature in stdlib as soon as it's stabilized,
584/// see https://github.com/rust-lang/rust/issues/90091
585fn split_array_ref<const N: usize, T>(slice: &[T]) -> (&[T; N], &[T]) {
586    let (a, b) = slice.split_at(N);
587    // SAFETY: a points to [T; N]? Yes it's [T] of length N (checked by split_at)
588    unsafe { (&*(a.as_ptr() as *const [T; N]), b) }
589}
